key_moments.py
async def select_key_moments(state):
    prompt = f"""전사 텍스트:
{state['transcription']}

임팩트 높은 장면 {N_CLIPS}개를 JSON으로:
[{{"start":0,"end":30,"reason":"..."}}]"""

    resp = await client.chat.completions.create(
        model="gemini-2.5-flash",
        messages=[{"role":"user","content":prompt}],
        response_format={"type":"json_object"},
    )
    clips = json.loads(resp.choices[0].message.content)
    return {"key_moments": clips["clips"]}


def build_graph():
    g = StateGraph(State)
    g.add_node("transcribe",    transcribe_audio)
    g.add_node("select",        select_key_moments)
    g.add_node("process_clip",  process_clip)

    # Send()로 장면 개수만큼 병렬 팬아웃
    g.add_conditional_edges(
        "select",
        lambda s: [
            Send("process_clip", {"clip": c, **s})
            for c in s["key_moments"]
        ],
    )
    g.add_edge(START, "transcribe")
    g.add_edge("transcribe", "select")
    return g.compile()

async def process_clip(state: State) -> State:
    clip   = state["clip"]
    job_id = state["job_id"]

    # TTS + 썸네일 병렬 생성
    tts_path, thumb_path = await asyncio.gather(
        generate_tts(clip["narration"], job_id),
        generate_thumbnail(clip["image_prompt"], job_id),
    )

    # 9:16 클립 합성
    clip_path = await make_clip(
        video_path = state["video_path"],
        start      = clip["start"],
        end        = clip["end"],
        audio      = tts_path,
    )

    # 자막 번인
    final_clip = await burn_subtitles(
        clip_path, clip["narration"]
    )

    return {
        "clips":      [final_clip],   # operator.add 누적
        "thumbnails": [thumb_path],
    }


async def concat_all(state: State) -> State:
    reel = await concat_clips(state["clips"])
    return {"highlight_reel": reel}

tts_maker.py
import edge_tts, subprocess
from pathlib import Path

VOICE = "ko-KR-SunHiNeural"
RATE  = "+10%"

async def generate_tts(
    text: str,
    output_path: Path,
) -> float:
    """TTS 생성 후 오디오 길이(초) 반환."""
    communicate = edge_tts.Communicate(
        text  = text,
        voice = VOICE,
        rate  = RATE,
    )
    await communicate.save(str(output_path))

    # 정확한 오디오 길이 측정
    result = subprocess.run(
        ["ffprobe", "-v", "error",
         "-show_entries", "format=duration",
         "-of", "default=nw=1:nk=1",
         str(output_path)],
        capture_output=True, text=True,
    )
    return float(result.stdout.strip())

from sse_starlette.sse import EventSourceResponse

@router.get("/jobs/{job_id}/events")
async def job_events(job_id: str):
    async def generator():
        while True:
            job = jobs.get(job_id)
            if not job:
                yield {"data": json.dumps({"error":"not found"})}
                return

            yield {
                "data": json.dumps({
                    "status":   job.status,
                    "step":     job.step,
                    "progress": job.progress,
                })
            }

            if job.status in ("done", "error"):
                return

            await asyncio.sleep(0.5)

    return EventSourceResponse(generator())

RAG

ChromaDB + GPT-4o 영상 질의응답

rag.py
import chromadb
from openai import OpenAI

client = OpenAI()
chroma = chromadb.PersistentClient(path="./chroma_db")

def index_transcript(job_id: str, text: str):
    col    = chroma.get_or_create_collection(job_id)
    chunks = [text[i:i+500] for i in range(0, len(text), 500)]
    embs   = client.embeddings.create(
        input=chunks, model="text-embedding-3-small"
    ).data
    col.add(
        documents  = chunks,
        embeddings = [e.embedding for e in embs],
        ids        = [f"chunk_{i}" for i in range(len(chunks))],
    )

def rag_query(job_id: str, question: str) -> str:
    col   = chroma.get_collection(job_id)
    q_emb = client.embeddings.create(
        input=[question], model="text-embedding-3-small"
    ).data[0].embedding

    results = col.query(query_embeddings=[q_emb], n_results=3)
    context = "\n".join(results["documents"][0])

    resp = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[
            {"role":"system","content":f"영상 내용:\n{context}"},
            {"role":"user",  "content": question},
        ],
    )
    return resp.choices[0].message.content

async def run_create_pipeline(job_id, script, out_dir):
    scenes = script.scenes  # List[SceneData]

    # ① 장면별 TTS + 이미지 병렬 생성
    scene_results = await asyncio.gather(*[
        _process_scene(scene, out_dir)
        for scene in scenes
    ])

    # ② 장면 클립 순차 합성
    clips = []
    for tts, img, dur in scene_results:
        clips.append(await _make_scene_clip(img, tts))

    # ③ 전체 concat → 자막 번인
    raw  = await concat_clips(clips)
    segs = _build_ass_segments(scenes, [d for _,_,d in scene_results])
    final = await burn_subtitles(raw, segs)

    # ④ 썸네일 = 첫 장면 이미지
    shutil.copy(scene_results[0][1], out_dir / "thumbnail.jpg")


async def _process_scene(scene, out):
    tts, img = await asyncio.gather(
        generate_tts(scene.narration,
                     out / f"tts_{scene.order}.mp3"),
        generate_image(scene.image_prompt,
                       out / f"img_{scene.order}.jpg"),
    )
    return tts, img, await get_duration(tts)

🗂️ 인메모리 Job 저장 영속성

서버 재시작 시 진행 중인 작업 상태가 초기화됩니다. 완료 결과물은 파일로 보존되지만 in-flight 작업은 복구 불가.

→ Redis 또는 SQLite로 Job 상태 영속화

⚡ 단일 서버 구조 확장성

동시 업로드 증가 시 CPU 집약 ffmpeg 작업이 이벤트 루프를 압박합니다. 현재는 단일 프로세스.

→ Celery + Redis 태스크 큐로 워커 수평 확장

🔁 파이프라인 재시도 불가 내결함성

LangGraph에 checkpointer가 없어 네트워크 오류 발생 시 처음부터 재실행해야 합니다. 부분 재개 불가.

→ MemorySaver / SqliteSaver로 노드 단위 체크포인트

💸 API 비용 의존 비용

gpt-image-1(high), OpenAI Whisper, GPT-4o 모두 호출당 과금. 대용량 배치 시 비용이 빠르게 증가합니다.

→ Faster-Whisper(로컬) · Stable Diffusion · Gemini Flash로 대체

📦 배포 환경 의존성 배포

ffmpeg(libass 포함), yt-dlp 등 로컬 바이너리 수동 설치 필요. 환경별 경로 차이로 재현성이 낮습니다.

→ Docker 컨테이너화로 의존성 봉인
